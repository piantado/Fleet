210  // this takes a collection T of elements t, and a function f
211  // which assigns them each a probability, and samples from them according
212  // to the probabilities in f. The probability that is returned is only the probability
213  // of selecting that *element* (index), not the probability of selecting anything equal to it
214  // (i.e. we defaultly don't double-count equal options). For that, use p_sample_eq below
215  // here z is the normalizer, z = sum_x f(x)
216  assert(z > 0 && "*** Cannot call sample with zero normalizer -- is s empty?");
217  double r = z * uniform();
218  for(auto& x : s) {
219 
220  auto fx = f(x);
221  if(std::isnan(fx)) continue; // treat as zero prob
222  assert(fx >= 0.0);
223  r -= fx;
224  if(r <= 0.0)
225  return std::make_pair(const_cast<t*>(&x), log(fx)-log(z));
226  }
