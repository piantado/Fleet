254  [[nodiscard]] virtual std::pair<this_t,double> propose() const override {
255

258  for(size_t i=0;i<factors.size();i++) {
259  should_propose[i] = flip(p_factor_propose);

261  should_propose[myrandom(should_propose.size())] = true; // always ensure one
262 
263  // now go through and propose to those factors
264  // (NOTE fb is always zero)
265  // NOTE: This is not great because it doesn't copy like we might want...

267  int idx = 0;
268  for(auto& [k,f] : factors) {
269  if(should_propose[idx]) {
270  auto [h, _fb] = f.propose();
271  x.factors[k] = h;
