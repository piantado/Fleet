233  assert(which_data == std::addressof(human_data));

236  int MX = -1;

502  virtual Vector hypothesis_prior(Matrix& myC) {
503  // take a matrix of counts (each row is a hypothesis)
504  // and return a prior for each hypothesis under my own X
505  // (If this was just a PCFG, which its not, we'd use something like lognormalize(C*proposal.getX()))
506 
507  Vector out(nhypotheses()); // one for each hypothesis
508 
509  // get the marginal probability of the counts via dirichlet-multinomial
510  Vector allA = logA.value.array().exp(); // translate [-inf,inf] into [0,inf]
511 
512  #pragma omp parallel for
513  for(auto i=0;i<myC.rows();i++) {
514 
515  double lp = 0.0;
516  size_t offset = 0; //
517  for(size_t nt=0;nt<grammar->count_nonterminals();nt++) { // each nonterminal in the grammar is a DM
518  size_t nrules = grammar->count_rules( (nonterminal_t) nt);
519  if(nrules > 1) { // don't need to do anything if only one rule (but must incremnet offset)
520  Vector a = eigenslice(allA,offset,nrules); // TODO: seqN doesn't seem to work with this eigen version
521  Vector c = eigenslice(myC.row(i),offset,nrules);
522  double n = a.sum(); assert(n > 0.0);
523  double c0 = c.sum();
524  if(c0 != 0.0) { // TODO: Check this...
525  // NOTE: std::lgamma here is not thread safe, so we use mylgamma defined in Numerics
526  lp += mylgamma(n+1) + mylgamma(c0) - mylgamma(n+c0)
527  + (c.array() + a.array()).array().lgamma().sum()
528  - (c.array()+1).array().lgamma().sum()
529  - a.array().lgamma().sum();
530  }
531  }
532  offset += nrules;
533  }
534 
535  #pragma omp critical
