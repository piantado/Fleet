474  virtual Vector hypothesis_prior(Matrix& myC) {
475  // take a matrix of counts (each row is a hypothesis)
476  // and return a prior for each hypothesis under my own X
477  // (If this was just a PCFG, which its not, we'd use something like lognormalize(C*proposal.getX()))
478 
479  Vector out(nhypotheses()); // one for each hypothesis
480 
481  // get the marginal probability of the counts via dirichlet-multinomial
482  Vector allA = logA.value.array().exp(); // translate [-inf,inf] into [0,inf]
483 
484  #pragma omp parallel for
485  for(auto i=0;i<myC.rows();i++) {
486 
487  double lp = 0.0;
488  size_t offset = 0; //
489  for(size_t nt=0;nt<grammar->count_nonterminals();nt++) { // each nonterminal in the grammar is a DM
490  size_t nrules = grammar->count_rules( (nonterminal_t) nt);
491  if(nrules > 1) { // don't need to do anything if only one rule (but must incremnet offset)
492  Vector a = eigenslice(allA,offset,nrules); // TODO: seqN doesn't seem to work with this eigen version
493  Vector c = eigenslice(myC.row(i),offset,nrules);
494  double n = a.sum(); assert(n > 0.0);
495  double c0 = c.sum();
496  if(c0 != 0.0) { // TODO: Check this...
497  // NOTE: std::lgamma here is not thread safe, but by my read it should be when the argument is positive
498  lp += std::lgamma(n+1) + std::lgamma(c0) - std::lgamma(n+c0)
499  + (c.array() + a.array()).array().lgamma().sum()
500  - (c.array()+1).array().lgamma().sum()
501  - a.array().lgamma().sum();
502  }
503  }
504  offset += nrules;
505  }
506 
507  out(i) = lp;
508  }
509 
510  return out;
