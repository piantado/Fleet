23  * A grammar stores all of the rules associated with any kind of nonterminal and permits us
24  * to sample as well as compute log probabilities.
25  * A grammar also is nwo required to store them in a fixed (sorted) order that is guaranteed
26  * not to change, adn that puts terminals first and (lower priority) high probability first
27  * as determined by Rule's sort order.

30  * to store them in the right places according to their return types (the index comes from
31  * the index of each type in FLEET_GRAMMAR_TYPES).
32  * The trees that Grammar generates use nt<T>() -> size_t to represent types, not the types
33  * themselves.
34  * The trees also use Primitive.op (a size_t) to represent operations

542  Node expand_from_integer(nonterminal_t nt, IntegerizedStack& is) const {
543  // this is a handy function to enumerate trees produced by the grammar.
544  // This works by encoding each tree into an integer using some "pairing functions"
545  // (see https://arxiv.org/pdf/1706.04129.pdf)
546  // Specifically, here is how we encode a tree: if it's a terminal,
547  // encode the index of the terminal in the grammar.
548  // If it's a nonterminal, encode the rule with a modular pairing function
549  // and then use Rosenberg-Strong encoding to encode each of the children.
550  // In this way, we can make a 1-1 pairing between trees and integers,
551  // and therefore store trees as integers in a reversible way
552  // as well as enumerate by giving this integers.
553 
554  // NOTE: for now this doesn't work when nt only has finitely many expansions/trees
555  // below it. Otherwise we'll get an assertion error eventually.
556 
557  enumerationidx_t numterm = count_terminals(nt);
558  if(is.get_value() < numterm) {
559  return makeNode(this->get_rule(nt, is.get_value())); // whatever terminal we wanted
560  }
561  else {
562 
563  is -= numterm; // remove this from is
564  auto ri = is.pop(count_nonterminals(nt));
565  Rule* r = this->get_rule(nt, ri+numterm); // shift index from terminals (because they are first!)
566  Node out = makeNode(r);
567  for(size_t i=0;i<r->N;i++) {
568  // note that this recurses on the z format -- so it makes a new IntegerizedStack for each child
569  out.set_child(i, expand_from_integer(r->type(i), i==r->N-1 ? is.get_value() : is.pop()) ) ;
570  }
571  return out;
572  }
573 
574  }

887  // here we find the neighbor indicated by which and expand it into the which'th neighbor

893  if(node.child(i).is_null()) {

895  if(which >= 0 and which < c) {

898  }
899  which -= c;
900  }
901  else { // otherwise we have to process that which
902  expand_to_neighbor(node.child(i), which);
