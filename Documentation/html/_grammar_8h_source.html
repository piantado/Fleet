889  Node simple_parse(std::string s) {
890  //PRINTN("Parsing ", s);
891 
892  // remove the lambda x. if its there
893  if(s.substr(0,3) == "\u03BBx.") s.erase(0,3);
894  // remove leading whitespace
895  while(s.at(0) == ' ' or s.at(0) == '\t') s.erase(0,1);
896  // remove trailing whitespace
897  while(s.at(s.size()-1) == ' ' or s.at(s.size()-1) == '\t') s.erase(s.size()-1,1);
898 
899  // use the above function to find chunks

901

904  assert(commas.size()==0 and close==-1);
905  auto r = this->get_rule(s);
906  return this->makeNode(r);
907  }
908  else if(close == open+1) { // special case of "f()"
909  return this->makeNode(get_rule(s)); // the whole string is what we want and its a terminal
910  }
911  else {
912  assert(close != -1);
913 
914  // recover the rule format -- no spaces, etc.
915  std::string fmt = s.substr(0,open) + "(%s";
916  for(auto& c : commas) {
917  UNUSED(c);
918  fmt += ",%s";
919  }

921

924  auto out = this->makeNode(r);

927  int ci=0;
928  for(auto& c : commas) {
929  auto child_string = s.substr(prev,c-prev);
930  out.set_child(ci,simple_parse(child_string));
931  prev = c+1;
932  ci++;
933  }
934 
935  // and the last child
936  auto child_string = s.substr(prev,close-prev);
937  out.set_child(ci,simple_parse(child_string));
938 
939  return out;
940  }
